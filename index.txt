        function setCookie(cname, cvalue, exdays) {
            var d = new Date();
            d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
            var expires = "expires="+d.toUTCString();
            document.cookie = cname + "=" + cvalue + ";" + expires + ";domain=.alphacoders.com;path=/;secure";
        }

        function getCookie(cname) {
            var name = cname + "=";
            var ca = document.cookie.split(';');
            for(var i = 0; i < ca.length; i++) {
                var c = ca[i];
                while (c.charAt(0) == ' ') {
                    c = c.substring(1);
                }
                if (c.indexOf(name) == 0) {
                    return c.substring(name.length, c.length);
                }
            }
            return "";
        }

                height = window.screen.height;
            }
            else {
                width = Math.floor(window.screen.width * ratio);
                height = Math.floor(window.screen.height * ratio);
            }

            if(parseInt(width.toString().split('').pop()) == 1 && parseInt(height.toString().split('').pop()) == 1) {
                width -= 1;

                    setTimeout(function () {
                        is_loading = 0;
                    }, 1000);
                });

            //If the page represent at least 2/5 of the page, used to update the url
            function mostlyVisible(element) {
                let box = element.getBoundingClientRect();
                let window_height = $(window).height();
                return ( box.top < ( window_height * (3/5) ) );
            }

            //Delete elements to free space/memory
            function deleteElement() {
                let containers = $(".page_container");

                if(containers.length > max_displayed) {
                    containers.slice(0, (containers.length-max_displayed)).each(function(){
                        const final_position =  $(window).scrollTop() - ($(this).height() + 10);

                        //Remove the container
                        $(this).css('height', $(this).height()).html("").remove();

                        //Update scroll position
                        if($(window).scrollTop() > (final_position * 1.05) || $(window).scrollTop() < (final_position * 0.95))
                        {
                            $(window).scrollTop(final_position);
                        }
                    });
                }
            }

            //Enable/Disable auto loading for the next pages
            $( "#big_container" ).on("click", ".auto-load-button-next", function() {
